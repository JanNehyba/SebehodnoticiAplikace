PROTECTED_COMBOS = [
    r"Nevěnování se výuce/Nepozornost při výuce\s*,\s*Neplnění školních povinností/nepřipravenost na výuku"

# Znak, kterým dočasně nahradíme vnitřní čárku
INTERNAL_COMMA_TOKEN = " ⟂ "

def protect_commas(text: str) -> str:
    s = str(text)
    for pat in PROTECTED_COMBOS:
        s = re.sub(pat, lambda m: m.group(0).replace(",", INTERNAL_COMMA_TOKEN), s)
    return s

def split_multi(value):
    """Rozdělí buňku na položky dle čárky, ale chrání speciální kombinace.
       Trimuje mezery, prázdné odstraňuje, vrací list řetězců."""
    if pd.isna(value):
        return []
    s = str(value).strip()
    if not s:
        return []
    s = protect_commas(s)
    parts = [p.strip() for p in s.split(",") if p.strip()]
    # vrátit původní čárku v chráněných kombinacích
    parts = [p.replace(INTERNAL_COMMA_TOKEN, ", ") for p in parts]
    # normalize whitespace
    parts = [' '.join(p.split()) for p in parts]
    return parts

def _normalize_items(items, rename_map):
    """Sjednotí položky podle mapy (case-insensitive), ořízne mezery,
       a deduplikuje při zachování původního pořadí."""
    mapping_lower = {str(k).lower().strip(): v for k, v in rename_map.items()}
    out, seen = [], set()
    for t in items:
        s = ' '.join(str(t).split())         # normalizace mezer
        key = s.lower().strip()
        s2 = mapping_lower.get(key, s)       # přejmenuj, pokud je v mapě
        if s2 not in seen:                   # deduplikace
            out.append(s2)
            seen.add(s2)
    return out

def _normalize_cell(cell, rename_map):
